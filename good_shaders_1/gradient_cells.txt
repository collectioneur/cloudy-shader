const mainFragment = tgpu["~unstable"].fragmentFn({
      in: { uv: d.vec2f },
      out: d.vec4f,
    })(({ uv }) => {
      {
        return d.vec4f(
          std.sin(uv.x * 3.0 + time.$) * 0.5 + 0.5,
          (std.sin(time.$ * 1.0 * uv.x) + 1.0) / 2.0,
          (uv.y * std.cos(time.$ * 3 * uv.y)) / 2.0 + 0.5,
          1.0
        );
      }
    });
